def is_QA(D):
    '''
    Returns true if the diagram D is quasi-alternating and false if it is unclear.
    '''
    if D.is_alternating():
        return True
    if D.exterior() in QA:
        return True
    PD=D.PD_code()
    det=D.determinant()
    for cros in PD:
        (a,b,c,d)=cros
        PD0=PD.copy()
        PD0.remove(cros)
        PD0copy=PD0.copy()
        PDinfty=PD0.copy()
        PDinftycopy=PD0.copy()
        for cro in PD0copy:
            for strand in cro:
                if strand==b:
                    PD0.remove(cro)
                    cro=tuple([a if x==b else x for x in list(cro)])
                    PD0.append(cro)
                if strand==d:
                    PD0.remove(cro)
                    cro=tuple([c if x==d else x for x in list(cro)])
                    PD0.append(cro)
        for cro in PDinftycopy:
            for strand in cro:
                if strand==c:
                    PDinfty.remove(cro)
                    cro=tuple([b if x==c else x for x in list(cro)])
                    PDinfty.append(cro)
                if strand==d:
                    PDinfty.remove(cro)
                    cro=tuple([a if x==d else x for x in list(cro)])    
                    PDinfty.append(cro)
        L0=snappy.Link(PD0)
        Linfty=snappy.Link(PDinfty)
        try:
            det0=L0.determinant(method='goeritz')
        except ValueError:
            try:
                det0=L0.determinant(method='wirt')
            except ValueError:
                try:
                    det0=L0.determinant(method='color')
                except ValueError:
                    pass
        try:
            detinfty=Linfty.determinant(method='goeritz')
        except ValueError:
            try:
                detinfty=Linfty.determinant(method='wirt')
            except ValueError:
                try:
                    detinfty=Linfty.determinant(method='color')
                except ValueError:
                    pass
        if det0+detinfty==det:
            L0.simplify('global')
            Linfty.simplify('global')
            w0=False
            if L0.PD_code()==[]:
                w0=True
            if L0.is_alternating()==True:
                w0=True
            if L0.exterior() in QA:
                w0=True
            if w0==True:
                if Linfty.PD_code()==[]:
                    return True
                if Linfty.is_alternating()==True:
                    return True
                if Linfty.exterior() in QA:
                    return True
    return False
